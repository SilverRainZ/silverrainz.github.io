зј–еҶҷдҫҝдәҺжү“еҢ…зҡ„ Makefile#

жҸҗзӨә

иҝҷжҳҜдёҖзҜҮиҝҒз§»иҮӘ Jekyll зҡ„ж–Үз« пјҢеҰӮжңүж јејҸй—®йўҳпјҢеҸҜеҲ° вӣә SilverRainZ/bullet еҸҚйҰҲ

Note: иҝҷзҜҮж–Үз« еҒҮи®ҫдҪ е·Із»ҸзҹҘйҒ“еҹәжң¬зҡ„ Makefile зј–еҶҷи§„еҲҷ

еүҚиЁҖ#

е®үиЈ…еҹәдәҺ make жһ„е»әзҡ„зЁӢеәҸеҹәжң¬дёҠе°ұжҳҜдёӨдёӘжӯҘйӘӨпјҡmake з„¶еҗҺ make installпјҢ еүҚиҖ…жҠҠзЁӢеәҸжҢүдҫқиө–е…ізі»зј–иҜ‘пјҢеҗҺиҖ…жҠҠж–ҮжЎЈгҖҒж•°жҚ®гҖҒзј–иҜ‘еҮәжқҘзҡ„дәҢиҝӣеҲ¶е®үиЈ…еҲ°зі»з»ҹдёӯгҖӮ зҪ‘з»ңдёҠе…ідәҺ GNU Make зҡ„ж•ҷзЁӢдёҚе°‘пјҢдҪҶдјјд№ҺйғҪжӯўдәҺгҖҢеҰӮдҪ•з”Ё Makefile иҮӘеҠЁзј–иҜ‘зЁӢеәҸгҖҚпјҲmakeпјүпјҢ иҖҢе…ідәҺз”Ё Makefile зј–еҶҷе®үиЈ…и„ҡжң¬пјҲmake installпјүзҡ„ж–Үз« еҚҙеҜҘеҜҘж— еҮ гҖӮ

2016-08-10: з»Ҹ зӣ–еӯҗ жҸҗйҶ’пјҢmake жң¬жқҘе°ұдёҚйҖӮеҗҲеҒҡиҝҷз§ҚдәӢжғ…пјҢдәҺжҳҜжүҚжңүдәҶ autotools иҝҷгҖҢжӣҙеҘҪгҖҚзҡ„жһ„е»әе·Ҙе…·гҖӮ

жңҖиҝ‘еңЁеҶҷ Srain зҡ„ж—¶еҖҷпјҢ з®—жҳҜж‘ёзҙўеҮәдәҶеҜ№дәҺ make install зҡ„жҜ”иҫғжӯЈзЎ®зҡ„еҶҷжі•пјҡ

йҰ–е…ҲпјҢеҜ№дәҺйЎ№зӣ®з”ҹжҲҗдә§зү©еҸӘжҳҜеҚ•дёӘеҸҜжү§иЎҢж–Үд»¶зҡ„жғ…еҶөдёӢпјҢзӣҙжҺҘеңЁ install зӣ®ж ҮйҮҢеҶҷ install -Dm755 xxx /usr/bin/xxx е°ұеҘҪдәҶгҖӮ дҪҶжҳҜе№¶йқһжүҖжңүйЎ№зӣ®йғҪеҸӘеҢ…еҗ«еҚ•дёӘеҸҜжү§иЎҢж–Үд»¶пјҢзЁӢеәҸеҸҜиғҪиҝҳеҢ…еҗ«дәҶ man ж–ҮжЎЈпјҢiconsпјҢ еӣҫзүҮпјҢй…ҚзҪ®ж–Үд»¶зӯүпјҢиҝҷдәӣйғҪиҰҒиў«е®үиЈ…еҲ°ж–Үд»¶зі»з»ҹзӣёеә”зҡ„дҪҚзҪ®дёҠгҖӮ

жҲ‘д»¬е…ҲеҒҮи®ҫйЎ№зӣ®зҡ„з»“жһ„еҰӮдёӢпјҢд»Јз ҒеҶҷдәҶд»Җд№ҲдёҚйҮҚиҰҒ~

е®үиЈ…еӣҫж Ү#

еҜ№дәҺеӣҫж ҮпјҢIcon Theme Specification[1] и§„е®ҡдәҶеӣҫж ҮеңЁж–Үд»¶зі»з»ҹдёҠзҡ„дҪҚзҪ®пјҢзЁӢеәҸеҸӘйңҖиҰҒж №жҚ®еӣҫж Үзҡ„еҗҚз§°пјҲеҚіж–Үд»¶еҗҚеҺ»жҺүжү©еұ•еҗҚпјү е’ҢеӨ§е°Ҹе°ұеҸҜд»ҘиҺ·еҫ—еӣҫж Үж–Үд»¶зҡ„и·Ҝеҫ„ пјҲеҪ“з„¶иҰҒеҖҹеҠ©еҗ„з§Қеә“еҮҪж•°пјҢжҜ”еҰӮ gtk зҡ„ gtk_image_new_from_icon_nameпјүпјҢ еӣ жӯӨжҲ‘д»¬еҸӘиҰҒе°Ҷеӣҫж Үж–Үд»¶еӨҚеҲ¶еҲ°еҜ№еә”зҡ„дҪҚзҪ®дёҠеҚіеҸҜгҖӮ

ж №жҚ®дёҠйқўзҡ„ specпјҢзЁӢеәҸеҜ»жүҫеӣҫж Үж—¶еә”иҜҘдҫқж¬ЎжЈҖжҹҘ $HOME/.iconsгҖҒ$XDG_DATA_DIRS/icons е’Ң /usr/share/pixmapsгҖӮ

еҸӮз…§ XDG Base Directory Specification[2] зңӢпјҢ еҪ“ $XDG_DATA_DIRS дёәз©әж—¶пјҢ$XDG_DATA_DIRS дјҡй»ҳи®Өдёә /usr/local/share/:/usr/share/ пјҲж„ҹи°ў csslayer жҢҮеҮәпјүгҖӮ

еӣ жӯӨжҠҠеӣҫж Үе®үиЈ…еңЁ /usr/share/pixmapsгҖҒ/usr/local/share/icons е’Ң /usr/share/icons дёӢйғҪжҳҜеҸҜиЎҢзҡ„пјҢArch Linux еҒҸеҗ‘дәҺе®үиЈ…еңЁжңҖеҗҺдёҖдёӘзӣ®еҪ•гҖӮ дәҺжҳҜе®үиЈ… еӨ§е°Ҹдёә 16x16 зҡ„еӣҫж Ү зҡ„и„ҡжң¬еҸҜд»Ҙиҝҷд№ҲеҶҷпјҡ

cd data/icons/16x16; \
    for png in *.png; do \
        install -Dm644 "$$png" \
            "$(DESTDIR)/usr/share/icons/hicolor/16x16/apps/$$png"; \
    done

иҝҷйҮҢе…ҲдёҚз®Ў $(DESTDIR) жҳҜд»Җд№ҲдёңиҘҝпјҢжҠҠе®ғеҪ“дҪңз©әеҸҳйҮҸеҚіеҸҜпјҡ

install -Dm644 "$$png" \
    "/usr/share/icons/hicolor/16x16/apps/$$png"; \

PREFIX#

йҷӨдәҶеӣҫж Үд№ӢеӨ–пјҢе…¶д»–зҡ„ж•°жҚ®ж–Үд»¶еә”иҜҘеҰӮдҪ•з»„з»Үпјҹ иҮіе°‘жҲ‘д»¬еә”иҜҘеҒҡеҲ°зҡ„жҳҜпјҡ

  • дҝқиҜҒзЁӢеәҸдёҖе®ҡиғҪжүҫеҲ°ж•°жҚ®ж–Үд»¶

  • дёҖе®ҡзЁӢеәҰдёҠе…Ғи®ёз”ЁжҲ·иҮӘе®ҡд№үе®үиЈ…зҡ„дҪҚзҪ®

GNU make жҸҗдҫӣдәҶ prefix зӯүеҸҳйҮҸзЎ®е®ҡеҗ„з§Қж–Үд»¶е®үиЈ…зҡ„дҪҚзҪ®[3]пјҡ

  • prefix жҳҜдёӢиҝ°еҸҳйҮҸзҡ„еүҚзјҖпјҢй»ҳи®Өзҡ„ prefix еҖјеә”иҜҘжҳҜ /usr/local

    • exec_prefix жҳҜдёӢиҝ°еҸҳйҮҸзҡ„еүҚзјҖпјҢйҖҡеёёе’Ң prefix зӣёзӯү

      • bindir е®үиЈ…еҸҜжү§иЎҢж–Үд»¶зҡ„дҪҚзҪ®пјҢе…¶еҖјеә”дёә $(exec_prefix)/bin

      • вҖҰ

    • datarootdir з”ЁжқҘе®үиЈ…еҸӘиҜ»зҡ„пјҢжһ¶жһ„ж— е…ізҡ„ж•°жҚ®ж–Үд»¶пјҢе…¶еҖјеә”дёә $(prefix)/share

    • sysconfdir з”ЁжқҘе®үиЈ…еҸӘиҜ»зҡ„й…ҚзҪ®ж–Үд»¶пјҢе…¶еҖјеә”дёә $(predix)/etc

    • вҖҰ

дёҠйқўеҲ—еҮәдәҶеҗ„з§Қз”ЁйҖ”зҡ„еҸҳйҮҸпјҢдҪҶдәӢе®һдёҠжҲ‘д»¬дёҚйңҖиҰҒжҠҠж•°жҚ®ж–Үд»¶еҲҶжҲҗйӮЈд№Ҳз»Ҷзҡ„зІ’еәҰгҖӮ еҜ№дәҺз®ҖеҚ•зҡ„йЎ№зӣ®пјҢеҸӘжңү prefix жҳҜеҝ…иҰҒзҡ„пјҢе…¶д»–и·Ҝеҫ„йғҪеҸҜд»Ҙ hardcodeгҖӮ

make install еҸҜд»Ҙиҝҷд№ҲеҶҷпјҲдёәдәҶе‘ҪеҗҚз»ҹдёҖпјҢprefix з”ЁеӨ§еҶҷпјүпјҡ

PREFIX = /usr/local

install:
    install -Dm755 "build/srain" "$(PREFIX)/bin/srain"
    cd data/pixmaps; \
        for png in *.png; do \
            install -Dm644 "$$png" \
                "$(PREFIX)/share/srain/pixmaps/$$png"; \
        done

ж”ҫзҪ®еҗ„з§Қж–Үд»¶зҡ„и§„иҢғжңүдәҶпјҢдҪҶзЁӢеәҸеә”иҜҘеҰӮдҪ•жүҫеҲ°д»–зҡ„ж•°жҚ®ж–Үд»¶е‘ўпјҹ з”Ё gcc зҡ„ -D еҸӮж•°еЈ°жҳҺдёҖдёӘе®ҸпјҢеңЁзј–иҜ‘зҡ„ж—¶еҖҷе‘ҠиҜүзЁӢеәҸзҡ„ prefixпјҡ

CC = gcc
CFLAGS = -O2 -Wall
DEFS = -DPACKAGE_DATA_DIR=\"$(PREFIX)\"

TARGET = build/srain

$(TARGET): srain.c
    $(CC) $(CFLAGS) $(DEFS) $^ -o $@

еңЁзЁӢеәҸдёӯдҪ е°ұеҸҜд»Ҙж №жҚ®иҝҷдёӘе®ҸеңЁиҺ·еҫ—дҪ зҡ„ж•°жҚ®ж–Үд»¶пјҡ

#ifndef PACKAGE_DATA_DIR
#define PACKAGE_DATA_DIR "/usr/local"
#endif

gchar *get_pixmap_path(const gchar *filename){
    gchar *path;

    path = g_build_filename(PACKAGE_DATA_DIR, "share",
            "srain", "pixmaps", filename, NULL);

    if (g_file_test(path, G_FILE_TEST_EXISTS)){
        return path;
    }

    g_free(path);
    return NULL;
}

жіЁж„ҸдёҠйқўзҡ„д»Јз ҒдҪҝз”ЁдәҶ glib еҮҪж•°еә“пјҢеҪ“жҢҮе®ҡ prefix дёә /usrпјҢ зЁӢеәҸдҫҝдјҡд»Һ /usr/share/srain/pixmaps йҮҢеҜ»жүҫеӣҫзүҮгҖӮ

иҮӘиЎҢзј–иҜ‘е®үиЈ…зҡ„зЁӢеәҸйҖҡеёёиў«е®үиЈ…еңЁ /usr/local, иҝҷд№ҹжҳҜ GNU жҺЁиҚҗзҡ„ prefixпјҢ Arch Linux зҡ„еҢ…зҡ„ prefix йҖҡеёёжҳҜ /usrгҖӮ

еҰӮдёҠдёҖз•Әи®ҫе®ҡеҗҺпјҢзЁӢеәҸз»ҸиҝҮзј–иҜ‘е’Ңе®үиЈ…еҗҺдҫҝеҸҜд»ҘиҝҗиЎҢжҢҮе®ҡзҡ„д»»ж„Ҹзӣ®еҪ•дёҠдәҶпјҢ дҪ д№ҹеҸҜд»ҘжҢҮе®ҡдёә $(PWD)/build ж–№дҫҝи°ғиҜ•гҖӮ

make PREFIX=/usr; make PREFIX=/usr install еҗҺпјҢдә§з”ҹзҡ„ж–Үд»¶еҰӮдёӢпјҡ

/usr/bin/srain
/usr/share/srain/pixmaps/srain-avatar.png
/usr/share/icons/hicolor/16x16/apps/srain-icon.png

make PREFIX=/home/la/tmp; make PREFIX=/home/la/tmp install еҲҷжҳҜпјҡ

/home/la/tmp/bin/srain
/home/la/tmp/share/srain/pixmaps/srain-avatar.png
/usr/share/icons/hicolor/16x16/apps/srain-icon.png

DESTDIR#

дёҠйқўзҡ„ make install зӣҙжҺҘе°Ҷеҗ„з§Қж–Үд»¶е®үиЈ…еңЁдәҶзӣ®зҡ„ж–Үд»¶зі»з»ҹдёҠпјҢеҰӮжһң Makefile еҶҷй”ҷзҡ„иҜқпјҢ еҸҜиғҪеҜ№зі»з»ҹйҖ жҲҗз ҙеқҸпјҢзӣҙжҺҘе®үиЈ…д№ҹдёҚеҲ©дәҺжү“еҢ…пјҢжӯЈзЎ®зҡ„еҒҡжі•жҳҜпјҢз”ұ make install еҫ—еҲ°зЁӢеәҸжүҖжңүж–Үд»¶зҡ„еҲ—иЎЁе’Ңи·Ҝеҫ„пјҢеҶҚз”ұеҢ…з®ЎзҗҶеҷЁжҠҠиҝҷдәӣж–Үд»¶е’Ңи·Ҝеҫ„еӯҳдёәиҪҜд»¶еҢ…пјҢ е®үиЈ…зҡ„ж—¶еҖҷж №жҚ®и·Ҝеҫ„жҠҠж–Үд»¶ж”ҫеҲ°еә”иҜҘж”ҫзҡ„дҪҚзҪ®пјҲиҝҷеӨ§жҰӮе°ұжҳҜ Staged InstallпјҹпјүгҖӮ пјҲиҝҷйҮҢж„ҹи°ўйқ’иӣҷиҖҒеёҲ рҹ‘Ө hexchain зҡ„жҢҮеҜјпјү

еҸҳйҮҸ DESTDIR[4] е°ұжҳҜз”ЁжқҘе®һзҺ° Staged Install зҡ„пјҢжҠҠд№ӢеүҚзҡ„ make install ж”№жҲҗиҝҷж ·пјҡ

PREFIX = /usr/local
install:
    install -Dm755 "build/srain" "$(DESTDIR)$(PREFIX)/bin/srain"
    cd data/pixmaps; \
        for png in *.png; do \
            install -Dm644 "$$png" \
                "$(DESTDIR)$(PREFIX)/share/srain/pixmaps/$$png"; \
        done

жіЁж„Ҹ DESTDIR еҸҳйҮҸеҸӘеә”иҜҘдҪңз”ЁеңЁ install йҳ¶ж®өпјҢmake PREFIX=/usr; make PREFIX=/usr DESTDIR=/tmp/ дјҡжҠҠжүҖжңүж–Үд»¶йғҪе®үиЈ…еңЁ /tmp дёӢпјҢ жүҖжңүзҡ„еҪұе“ҚйғҪиў«йҷҗеҲ¶еңЁиҜҘзӣ®еҪ•еҶ…гҖӮиҝҷж¬Ўз”ҹжҲҗзҡ„ж–Үд»¶еә”иҜҘжҳҜпјҡ

/tmp/usr/bin/srain
/tmp/usr/share/srain/pixmaps/srain-avatar.png
/tmp/usr/share/icons/hicolor/16x16/apps/srain-icon.png

Configure#

дёҠйқўзҡ„ Makefile жңүеӨ„дёҚдјҳйӣ…зҡ„ең°ж–№жҳҜпјҢmake е’Ң make install зҡ„ж—¶еҖҷеҝ…йЎ»жҢҮе®ҡзӣёеҗҢзҡ„ PREFIXпјҢ дёҚз„¶е®үиЈ…еҗҺзҡ„зЁӢеәҸиӮҜе®ҡжҳҜиҝҗиЎҢдёҚдәҶзҡ„пјҢиҖҢ make жң¬иә«е№¶дёҚиғҪи§ЈеҶіиҝҷдёӘй—®йўҳпјҢеӣ дёә make жҳҜгҖҢж— зҠ¶жҖҒгҖҚзҡ„гҖӮ

иҝҷйҮҢ[5]жҸҗдҫӣдәҶдёҖдёӘи„ҡжң¬жқҘи®©и§ЈеҶіиҝҷдёӘй—®йўҳпјҢе°Ҷ Makefile ж”№еҗҚдёә Makefile.inпјҢ иҝҗиЎҢ ./configure --prefix=xxx жқҘиҺ·еҫ—дёҖдёӘжӢҘжңүжҢҮе®ҡ prefix зҡ„ MakefileпјҢ иҝҷж ·е°ұеҸҜд»ҘдёҚз”ЁжҜҸж¬Ўж•І make йғҪиҫ“е…Ҙ PREFIX=xxx дәҶгҖӮ

дәҺжҳҜеӨ§е®¶йғҪеҺ»з”Ё autotools дәҶ

#!/bin/sh

prefix=/usr/local

for arg in "$@"; do
    case "$arg" in
    --prefix=*)
        prefix=`echo $arg | sed 's/--prefix=//'`
        ;;

    --help)
        echo 'usage: ./configure [options]'
        echo 'options:'
        echo '  --prefix=<path>: installation prefix'
        echo 'all invalid options are silently ignored'
        exit 0
        ;;
    esac
done

echo 'generating makefile ...'
echo "PREFIX = $prefix" >Makefile
cat Makefile.in >>Makefile
echo 'configuration complete, type make to build.'

еҰӮдёҠпјҢжү§иЎҢ ./configure --prefix=/usr е°ұдјҡжҠҠ Makefile.in еӨҚеҲ¶дёә MakefileпјҢе№¶еңЁ Makefile жңҖеүҚйқўеҠ дёҠдёҖеҸҘ PREFIX = /usrпјҲе®һйҷ…ж“ҚдҪңйЎәеәҸжҳҜеҸҚиҝҮжқҘзҡ„дҪ д»¬жҮӮе°ұеҘҪпјүгҖӮ
